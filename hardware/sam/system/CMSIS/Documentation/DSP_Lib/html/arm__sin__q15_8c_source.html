00103 q15_t arm_sin_q15( 
00104   q15_t x) 
00105 { 
00106   q31_t sinVal;                                  /* Temporary variables output */ 
00107   q15_t *tablePtr;                               /* Pointer to table */ 
00108   q15_t fract, in, in2;                          /* Temporary variables for input, output */ 
00109   q31_t wa, wb, wc, wd;                          /* Cubic interpolation coefficients */ 
00110   q15_t a, b, c, d;                              /* Four nearest output values */ 
00111   q15_t fractCube, fractSquare;                  /* Temporary values for fractional value */ 
00112   q15_t oneBy6 = 0x1555;                         /* Fixed point value of 1/6 */ 
00113   q15_t tableSpacing = TABLE_SPACING_Q15;        /* Table spacing */ 
00114   int32_t index;                                 /* Index variable */ 
00115  
00116   in = x; 
00117  
00118   /* Calculate the nearest index */ 
00119   index = (int32_t) in / tableSpacing; 
00120  
00121   /* Calculate the nearest value of input */ 
00122   in2 = (q15_t) ((index) * tableSpacing); 
00123  
00124   /* Calculation of fractional value */ 
00125   fract = (in - in2) << 8; 
00126  
00127   /* fractSquare = fract * fract */ 
00128   fractSquare = (q15_t) ((fract * fract) >> 15); 
00129  
00130   /* fractCube = fract * fract * fract */ 
00131   fractCube = (q15_t) ((fractSquare * fract) >> 15); 
00132  
00133   /* Initialise table pointer */ 
00134   tablePtr = (q15_t *) & sinTableQ15[index]; 
00135  
00136   /* Cubic interpolation process */ 
00137   /* Calculation of wa */ 
00138   /* wa = -(oneBy6)*fractCube + (fractSquare >> 1u) - (0x2AAA)*fract; */ 
00139   wa = (q31_t) oneBy6 *fractCube; 
00140   wa += (q31_t) 0x2AAA * fract; 
00141   wa = -(wa >> 15); 
00142   wa += ((q31_t) fractSquare >> 1u); 
00143  
00144   /* Read first nearest value of output from the sin table */ 
00145   a = *tablePtr++; 
00146  
00147   /* sinVal = a * wa */ 
00148   sinVal = a * wa; 
00149  
00150   /* Calculation of wb */ 
00151   wb = (((q31_t) fractCube >> 1u) - (q31_t) fractSquare) - 
00152        (((q31_t) fract >> 1u) - 0x7FFF); 
00153  
00154   /* Read second nearest value of output from the sin table */ 
00155   b = *tablePtr++; 
00156  
00157   /*      sinVal += b*wb */ 
00158   sinVal += b * wb; 
00159  
00160  
00161   /* Calculation of wc */ 
00162   wc = -(q31_t) fractCube + fractSquare; 
00163   wc = (wc >> 1u) + fract; 
00164  
00165   /* Read third nearest value of output from the sin table */ 
00166   c = *tablePtr++; 
00167  
00168   /*      sinVal += c*wc */ 
00169   sinVal += c * wc; 
00170  
00171   /* Calculation of wd */ 
00172   /* wd = (oneBy6)*fractCube - (oneBy6)*fract; */ 
00173   fractCube = fractCube - fract; 
00174   wd = ((q15_t) (((q31_t) oneBy6 * fractCube) >> 15)); 
00175  
00176   /* Read fourth nearest value of output from the sin table */ 
00177   d = *tablePtr++; 
00178  
00179   /* sinVal += d*wd; */ 
00180   sinVal += d * wd; 
00181  
00182   /* Return the output value in 1.15(q15) format */ 
00183   return ((q15_t) (sinVal >> 15u));
